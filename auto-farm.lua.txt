local ws    = game:GetService("Workspace")
local rs    = game:GetService("RunService")
local rstor = game:GetService("ReplicatedStorage")
local plrs  = game:GetService("Players")
local plr   = plrs.LocalPlayer
local farms = ws.Farms

local weaponRemote = rstor.Events.Weapon

local weapon_args = {
    a1 = "self",
    a2 = "219629d5067eddcdce55ed7968e9b53f"
}

-- TOGGLE (start OFF)
local enabled = false
local holding = false

-- anti-afk
local VirtualUser = game:GetService("VirtualUser")
game:GetService("Players").LocalPlayer.Idled:Connect(function()
    VirtualUser:CaptureController()
    VirtualUser:ClickButton2(Vector2.new())
end)

-- press F to toggle ON/OFF
local UserInputService = game:GetService("UserInputService")
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if input.KeyCode == Enum.KeyCode.F and not gameProcessed then
        enabled = not enabled
        print("Auto farm is now:", enabled and "ON" or "OFF")
    end
end)

-- hold LMB function
local function holdLMB()
    if holding then return end
    holding = true
    VirtualUser:CaptureController()
    VirtualUser:ClickButton1(Vector2.new(), true) -- press down
end

local function releaseLMB()
    if not holding then return end
    holding = false
    VirtualUser:ClickButton1(Vector2.new(), false) -- release
end

for i,v in pairs(farms:GetChildren()) do
    if tostring(v.Owner.Value) == tostring(plr.Name) then
        rs.RenderStepped:Connect(function()
            if enabled then
                plr.Character.HumanoidRootPart.CFrame = v.Chicken.CFrame
                holdLMB()
            else
                releaseLMB()
            end
        end)
    end
end
